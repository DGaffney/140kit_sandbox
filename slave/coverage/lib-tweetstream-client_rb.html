100     # Make a call to the statuses/filter method of the Streaming API,
101     # you may provide <tt>:follow</tt>, <tt>:track</tt> or both as options
102     # to follow the tweets of specified users or track keywords. This
103     # method is provided separately for cases when it would conserve the
104     # number of HTTP connections to combine track and follow.
105     def filter(query_params = {}, &block)
106       [:follow, :track, :locations].each do |param|
107         if query_params[param].is_a?(Array)
108           query_params[param] = query_params[param].collect{|q| q.to_s}.join(',')
109         elsif query_params[param]
110           query_params[param] = query_params[param].to_s
111         end
112       end
113       start('statuses/filter', query_params.merge(:method => :post), &block)
114     end

187     def start(path, query_parameters = {}, &block) #:nodoc:
188       method = query_parameters.delete(:method) || :get
189       delete_proc = query_parameters.delete(:delete) || self.on_delete
190       limit_proc = query_parameters.delete(:limit) || self.on_limit
191       error_proc = query_parameters.delete(:error) || self.on_error
192       
193       uri = method == :get ? build_uri(path, query_parameters) : build_uri(path)
194       
195       EventMachine::run {
196         EventMachine::add_periodic_timer(@periodic_timer[:seconds]) { @periodic_timer[:proc].call } if @periodic_timer
197         @stream = Twitter::JSONStream.connect(
198           :path => uri,
199           :auth => "#{URI.encode self.username}:#{URI.encode self.password}",
200           :method => method.to_s.upcase,
201           :content => (method == :post ? build_post_body(query_parameters) : ''),
202           :user_agent => 'TweetStream'
203         )
204         
205         @stream.each_item do |item|
206           raw_hash = @parser.decode(item)
207           
208           unless raw_hash.is_a?(::Hash)
209             error_proc.call("Unexpected JSON object in stream: #{item}")
210             next
211           end
212           
213           hash = TweetStream::Hash.new(raw_hash) # @parser.parse(item)
214           
215           if hash[:delete] && hash[:delete][:status]
216             delete_proc.call(hash[:delete][:status][:id], hash[:delete][:status][:user_id]) if delete_proc.is_a?(Proc)
217           elsif hash[:limit] && hash[:limit][:track]
218             limit_proc.call(hash[:limit][:track]) if limit_proc.is_a?(Proc)
219           elsif hash[:text] && hash[:user]
220             @last_status = TweetStream::Status.new(hash)
221             
222             # Give the block the option to receive either one
223             # or two arguments, depending on its arity.
224             case block.arity
225               when 1
226                 yield @last_status
227               when 2
228                 yield @last_status, self
229             end
230           end
231         end
232         
233         @stream.on_error do |message|
234           error_proc.call(message) if error_proc.is_a?(Proc)
235         end
236         
237         @stream.on_max_reconnects do |timeout, retries|
238           raise TweetStream::ReconnectError.new(timeout, retries)
239         end
240       }
241     end
