 16     #TODO: split this into multiple functions??

 18     # tweet and user counts
 19     loop do
 20       # this set up favors distribution of work
 21       datasets = Dataset.find_all({:scrape_finished => true, :tweets_count => nil, :users_count => nil})
 22       break if datasets.empty?
 23       # pick a random one
 24       dataset = datasets[rand(datasets.length-1)]
 25       dataset.tweets_count = Tweet.count({:dataset_id => dataset.id})
 26       dataset.users_count = User.count({:dataset_id => dataset.id})
 27       # delete empty datasets
 28       if dataset.tweets_count == 0
 29         # there really should only be one curation at this point
 30         for curation in dataset.curations
 31           if curation.datasets.length > 1
 32             dataset.destroy
 33           else
 34             self.generate_scrape_failed_email(curation)
 35             dataset.destroy
 36             curation.destroy
 37             Database.submit("delete from curations_datasets where dataset_id = '#{dataset.id}' and curation_id = '#{curation.id}';")
 38           end
 39         end
 40       else
 41         dataset.save
 42       end
 43     end

 46     curations = Curation.find_all({:analyzed => false}).reject {|c| c.datasets.collect {|d| d.scrape_finished }.include?(false) }

 48       # if the num of finished metadatas == total metadatas and total > 0

139           new_terms = new_terms - existing_terms
140           new_terms = new_terms - @@words
